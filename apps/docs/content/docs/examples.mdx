## Batch Processing

Process multiple videos with the same template.

```typescript
import { promises as fs } from 'fs';

async function batchProcessVideos(inputDir: string, outputDir: string) {
  const files = await fs.readdir(inputDir);
  const videoFiles = files.filter(file => file.endsWith('.mp4'));
  
  for (const file of videoFiles) {
    const inputPath = `${inputDir}/${file}`;
    const outputPath = `${outputDir}/processed-${file}`;
    
    const timeline = new Timeline()
      .addVideo(inputPath)
      
      // Add consistent branding
      .addImage('brand-logo.png', {
        position: { x: '85%', y: '15%' },
        scale: 0.2,
        duration: 'full'
      })
      
      // Add intro text
      .addText('Branded Content', {
        position: { x: 'center', y: '90%' },
        style: {
          fontSize: 32,
          color: '#ffffff',
          fontFamily: 'Arial Bold',
          backgroundColor: 'rgba(0,0,0,0.7)',
          padding: 10
        },
        startTime: 0,
        duration: 3
      });
    
    const command = timeline.getCommand(outputPath, {
      quality: 'high',
      codec: 'h264'
    });
    
    console.log(`Processing: ${file}`);
    console.log(`Command: ${command}`);
    
    // Execute command using your preferred method
    // await executeFFmpeg(command);
  }
}

// Usage
// await batchProcessVideos('./input-videos', './output-videos');
```

<Callout type="tip">
  **Performance Tip**: For batch processing, consider using the `FFmpegExecutor` with parallel processing limits to avoid overwhelming your system.
</Callout>
